#金额场组件

        hui-widget-attr='{"decimalLen":"3","maxLength":"10","minus":"false"}'>
</div>
                        
/**
 * setAmount: 设置金额
 * getAmount: 获得金额的字符串
 * getFloatAmt: 获得浮点型的金额值
 */

//amount1.setAmount(String(-100));
console.info(amount1.getAmount());
